CSV & JSON Handling in Python

Python provides built-in modules for working with CSV and JSON data formats, making it easy to read, write, and manipulate structured data. These formats are commonly used for data exchange, configuration, and storage in modern applications.

CSV Fundamentals

Introduction to CSV

CSV (Comma-Separated Values) is a simple file format used to store tabular data, such as a spreadsheet or database. Each line of the file is a data record, and each record consists of fields separated by commas (or other delimiters). 

# Example CSV data
"""
name,age,city
Alice,30,New York
Bob,25,San Francisco
Charlie,35,Seattle
"""

Basic CSV Reading

Python's csv module provides tools for working with CSV files. The basic reader creates lists for each row. 

import csv

# Reading CSV with basic reader
with open('data.csv', 'r', newline='') as file:
    reader = csv.reader(file)
    header = next(reader)  # Get header row
    
    for row in reader:
        print(f"Name: {row[0]}, Age: {row[1]}, City: {row[2]}")
        
# Reading CSV with named columns (DictReader)
with open('data.csv', 'r', newline='') as file:
    reader = csv.DictReader(file)
    
    for row in reader:
        print(f"Name: {row['name']}, Age: {row['age']}, City: {row['city']}")

Basic CSV Writing

Writing CSV files is equally straightforward, using either the basic writer or DictWriter. 

import csv

# Writing CSV with basic writer
with open('output.csv', 'w', newline='') as file:
    writer = csv.writer(file)
    writer.writerow(['name', 'age', 'city'])  # Header
    writer.writerow(['Alice', 30, 'New York'])
    writer.writerow(['Bob', 25, 'San Francisco'])
    
    # Write multiple rows at once
    more_data = [
        ['Charlie', 35, 'Seattle'],
        ['David', 28, 'Boston']
    ]
    writer.writerows(more_data)
    
# Writing CSV with named columns (DictWriter)
with open('output_dict.csv', 'w', newline='') as file:
    fieldnames = ['name', 'age', 'city']
    writer = csv.DictWriter(file, fieldnames=fieldnames)
    
    writer.writeheader()  # Write header row
    writer.writerow({'name': 'Alice', 'age': 30, 'city': 'New York'})
    writer.writerow({'name': 'Bob', 'age': 25, 'city': 'San Francisco'})

Advanced CSV Techniques

Custom Delimiters and Formats

While CSV files traditionally use commas as delimiters, the CSV module can handle various separators and formats. 

import csv

# Reading TSV (Tab-Separated Values)
with open('data.tsv', 'r', newline='') as file:
    reader = csv.reader(file, delimiter='\t')
    for row in reader:
        print(row)
        
# Writing with semicolon delimiter
with open('data_euro.csv', 'w', newline='') as file:
    writer = csv.writer(file, delimiter=';')
    writer.writerow(['name', 'age', 'city'])
    writer.writerow(['Alice', 30, 'Paris'])
    
# CSV dialect for custom formatting
csv.register_dialect('pipes', delimiter='|', quoting=csv.QUOTE_NONE, escapechar='\\')

with open('data_pipes.csv', 'w', newline='') as file:
    writer = csv.writer(file, dialect='pipes')
    writer.writerow(['name', 'age', 'city'])
    writer.writerow(['Alice', 30, 'New York|City'])  # Escapes the pipe in the city name

Handling Complex Data

CSV files can contain quotes, special characters, and multi-line fields. The CSV module provides options for handling these cases. 

import csv

# Handling quoted fields and special characters
with open('complex_data.csv', 'w', newline='') as file:
    writer = csv.writer(file, quoting=csv.QUOTE_NONNUMERIC)
    writer.writerow(['name', 'description', 'price'])
    writer.writerow(['Product A', 'Best item, "highly rated"', 19.99])
    writer.writerow(['Product B', 'Includes special characters: %, $, &', 24.50])
    
# Reading with different quoting options
with open('complex_data.csv', 'r', newline='') as file:
    reader = csv.reader(file, quoting=csv.QUOTE_NONNUMERIC)
    for row in reader:
        print(row)

Error Handling and Validation

Real-world CSV files can contain errors or inconsistencies that need to be handled gracefully. 

import csv

# Handling errors when reading CSV
try:
    with open('data.csv', 'r', newline='') as file:
        reader = csv.reader(file)
        for row in reader:
            if len(row) < 3:  # Check if row has all expected fields
                print(f"Warning: Incomplete row: {row}")
                continue
                
            try:
                age = int(row[1])  # Try to convert age to integer
            except ValueError:
                print(f"Warning: Invalid age in row: {row}")
                continue
                
            print(f"Valid row: {row}")
except FileNotFoundError:
    print("CSV file not found")
except csv.Error as e:
    print(f"CSV parsing error: {e}")

JSON Fundamentals

Introduction to JSON

JSON (JavaScript Object Notation) is a lightweight data interchange format that is easy for humans to read and write, and easy for machines to parse and generate. It's widely used for APIs, configuration files, and data storage. 

# Example JSON data
"""
{
  "name": "Alice",
  "age": 30,
  "is_student": false,
  "courses": ["Python", "Data Science", "Machine Learning"],
  "address": {
    "city": "New York",
    "zip": "10001"
  }
}
"""

Basic JSON Operations

Python's json module provides methods to encode (dump) Python objects to JSON strings and decode (load) JSON strings back to Python objects. 

import json

# Creating JSON from Python objects
data = {
    "name": "Alice",
    "age": 30,
    "is_student": False,
    "courses": ["Python", "Data Science", "Machine Learning"],
    "address": {
        "city": "New York",
        "zip": "10001"
    }
}

# Convert Python object to JSON string
json_string = json.dumps(data)
print(json_string)

# Convert Python object to JSON string with pretty formatting
json_pretty = json.dumps(data, indent=2, sort_keys=True)
print(json_pretty)

# Write JSON to a file
with open("data.json", "w") as file:
    json.dump(data, file, indent=4)

# Parse JSON string to Python object
parsed_data = json.loads(json_string)
print(parsed_data["name"])  # Alice
print(parsed_data["courses"][0])  # Python

# Read JSON from a file
with open("data.json", "r") as file:
    loaded_data = json.load(file)
    print(loaded_data["address"]["city"])  # New York

Advanced JSON Techniques

Custom Encoders and Decoders

JSON natively supports only certain Python types (dict, list, str, int, float, bool, None). For custom objects or unsupported types, you need custom encoders and decoders. 

import json
from datetime import datetime, date

# Custom JSON encoder
class CustomEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        elif isinstance(obj, date):
            return obj.isoformat()
        elif hasattr(obj, '__dict__'):  # For custom classes
            return obj.__dict__
        return super().default(obj)

# Data with datetime
data_with_dates = {
    "name": "Alice",
    "created_at": datetime.now(),
    "birthdate": date(1990, 5, 15)
}

# Use custom encoder
json_string = json.dumps(data_with_dates, cls=CustomEncoder)
print(json_string)

# Custom JSON decoder (for dates)
def date_decoder(dct):
    for key, value in dct.items():
        if isinstance(value, str):
            try:
                # Try to parse as datetime
                dct[key] = datetime.fromisoformat(value)
            except ValueError:
                try:
                    # Try to parse as date
                    dct[key] = date.fromisoformat(value)
                except ValueError:
                    pass
    return dct

# Decode JSON with custom handler
parsed_data = json.loads(json_string, object_hook=date_decoder)
print(type(parsed_data["created_at"]))  #

Error Handling and Validation

When working with external JSON data, it's important to handle errors and validate the structure. 

import json

# Error handling for invalid JSON
invalid_json = '{"name": "Alice", "age": 30,'  # Missing closing brace

try:
    parsed = json.loads(invalid_json)
except json.JSONDecodeError as e:
    print(f"Error parsing JSON: {e}")
    print(f"Error at position {e.pos}: {invalid_json[e.pos-10:e.pos+10]}")
    
# JSON validation function
def validate_user_json(json_data):
    required_fields = ["name", "email", "age"]
    
    try:
        data = json.loads(json_data) if isinstance(json_data, str) else json_data
    except json.JSONDecodeError:
        return False, "Invalid JSON format"
        
    if not isinstance(data, dict):
        return False, "JSON root must be an object"
        
    # Check required fields
    missing = [field for field in required_fields if field not in data]
    if missing:
        return False, f"Missing required fields: {', '.join(missing)}"
        
    # Validate field types
    if not isinstance(data.get("name"), str):
        return False, "Name must be a string"
        
    if not isinstance(data.get("email"), str):
        return False, "Email must be a string"
        
    if not isinstance(data.get("age"), (int, float)):
        return False, "Age must be a number"
        
    return True, "Valid user JSON"

CSV & JSON with Pandas

Using Pandas for Tabular Data

Pandas is a powerful data analysis library that provides more advanced functions for working with CSV and JSON data, especially for larger datasets and analytical tasks. 

import pandas as pd

# Reading CSV with pandas
df = pd.read_csv('data.csv')
print(df.head())  # View first few rows

# Reading CSV with custom options
df = pd.read_csv('data.csv',
                 skiprows=2,  # Skip first 2 rows
                 usecols=['name', 'age'],  # Select specific columns
                 dtype={'age': float},  # Set column data types
                 na_values=['N/A', 'unknown'])  # Handle missing values

# Writing CSV with pandas
df.to_csv('output.csv', 
          index=False,  # Don't write row indices
          columns=['name', 'age', 'city'],  # Select columns to write
          float_format='%.2f')  # Format floats to 2 decimal places

# Reading JSON with pandas
df = pd.read_json('data.json')

# Reading JSON with specific options
df = pd.read_json('data.json',
                  orient='records',  # JSON format style
                  lines=True)  # For JSON Lines format (one object per line)

# Writing JSON with pandas
df.to_json('output.json',
           orient='records',  # Array of objects
           date_format='iso')  # Format dates as ISO strings

Best Practices

CSV Best Practices

  • Always use the newline='' parameter when opening CSV files to ensure consistent behavior across platforms
  • Use DictReader and DictWriter for more readable code when handling named columns
  • Consider using pandas for large CSV files or complex data transformations
  • Explicitly specify encoding when dealing with international data (e.g., encoding='utf-8')
  • Be cautious with file paths - use os.path for cross-platform compatibility
  • Implement proper error handling for file operations and data validation

JSON Best Practices

  • Use indent parameter for human-readable JSON output in files, logs, or debugging
  • Implement custom encoders/decoders for non-JSON types like datetime
  • Always handle JSON parsing errors, especially with external data
  • Validate JSON against an expected schema when working with user input or APIs
  • Consider using ujson or orjson libraries for performance-critical applications
  • Be careful with large JSON files - consider streaming techniques or specialized libraries
# Example of robust CSV reading
import csv
import os

def read_csv_safely(filepath, encoding='utf-8'):
    if not os.path.exists(filepath):
        raise FileNotFoundError(f"CSV file not found: {filepath}")
        
    try:
        with open(filepath, 'r', newline='', encoding=encoding) as file:
            reader = csv.DictReader(file)
            return list(reader)
    except UnicodeDecodeError:
        print(f"Encoding error. Trying with 'latin-1' encoding...")
        with open(filepath, 'r', newline='', encoding='latin-1') as file:
            reader = csv.DictReader(file)
            return list(reader)
    except Exception as e:
        print(f"Error reading CSV: {e}")
        return []

# Example of robust JSON handling
import json

def save_json_safely(data, filepath, indent=2):
    # Ensure directory exists
    directory = os.path.dirname(filepath)
    if directory and not os.path.exists(directory):
        os.makedirs(directory)
        
    try:
        with open(filepath, 'w', encoding='utf-8') as file:
            json.dump(data, file, indent=indent, ensure_ascii=False)
        return True
    except Exception as e:
        print(f"Error saving JSON: {e}")
        return False
